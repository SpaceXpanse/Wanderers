	ExtMFD.cpp
	MFDWindow.cpp
	MFDWindow.h
	ExtMFD.rc
)

	D3D9Client
)

	${SpaceXpanseTgt}

)

	LIBRARY	
	DESTINATION ${SPACEXPANSE_INSTALL_PLUGIN_DIR}
